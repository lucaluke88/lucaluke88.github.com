Implementazione nativa di Pick di lodash

Sep 5, 2021 

export const pick = <T, K extends keyof T>(o: T, propertyNames: K[]) =>
    Object.fromEntries(
      Object.entries(o).filter(([key]) => (propertyNames as string[]).includes(key))
    ) as Pick<T, K>;
